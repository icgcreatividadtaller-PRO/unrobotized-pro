async function checkAndExecute() { const text = document.getElementById('inputText').value; if(!text) return; try { const response = await fetch('/api/check-limit'); const data = await response.json(); if (data.blocked || missionsLeft <= 0) { showPaywall(); return; } } catch (e) { console.error("Error en el guardia"); } try { const res = await fetch('/api/optimize', { method: 'POST', body: JSON.stringify({ text: text, mode: currentMode }), headers: {'Content-Type': 'application/json'} }); const result = await res.json(); // VALIDACIÓN CRÍTICA: Si no hay texto, mostramos el error del servidor if (result.humanizedText) { document.getElementById('outputText').innerText = result.humanizedText; missionsLeft--; localStorage.setItem('unrobotized_misiones', missionsLeft); document.getElementById('mission-count').innerText = `${missionsLeft}/3 DISPONIBLES`; addLogEntry(result.humanizedText); } else { // Si el servidor mandó un error, lo mostramos aquí document.getElementById('outputText').innerText = "ERROR NEURAL: " + (result.error || "Fallo desconocido"); } } catch (error) { document.getElementById('outputText').innerText = "ERROR DE CONEXIÓN: Verifica tu API Key en Vercel."; } }
